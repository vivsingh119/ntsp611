create Database EXAM_DATA;

use EXAM_DATA;


Step-3 (Create Table)
======================
CREATE TABLE `EXAM_RESULT1` 
(
  `id`         		bigint(20) NOT NULL      AUTO_INCREMENT,
  `dob`   			timestamp  NOT NULL      DEFAULT CURRENT_TIMESTAMP,
  `Semester`    	int(11)                  DEFAULT NULL,
  `percentage`      float             		 DEFAULT NULL,

  PRIMARY KEY (`id`)
);
